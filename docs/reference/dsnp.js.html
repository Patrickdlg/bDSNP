'use strict';

const EventEmitter = require('events');

class DSNP extends EventEmitter{
  /**
   * Create a plugin.
   * @constructor
   * @param {Options} options
   */
  constructor(options) {
    super();
    this.network = options.network
    this.init();
  }

  init(){
     console.log('connecting to: %s', this.network);
  }
}

module.exports = DSNP
